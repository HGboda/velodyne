project(ROAD_TIME)

# ========================================
# Compiler definitions
# ========================================
add_definitions(
    -DROAD_TIME_EXPORTS
)

# ========================================
# List of sources
# ========================================
set(
    PROJECT_SRCS
    ./src/road_time.c
)

# ========================================
# Build a library
# ========================================
pacpus_add_library(
    ${PROJECT_NAME} SHARED
    ${PROJECT_SRCS}
)

# ========================================
# Install
# ========================================
pacpus_install(${PROJECT_NAME})

# ========================================
# Folder
# ========================================
pacpus_folder(${PROJECT_NAME} "libraries")
